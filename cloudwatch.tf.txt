# Specify the provider
provider "aws" {
  region = "us-east-1"  # Change to your preferred region
}
 
# Create a CloudWatch Log Group
resource "aws_cloudwatch_log_group" "example" {
  name              = "example-log-group"
  retention_in_days = 14  # Set retention period
}
 
# Create a CloudWatch Alarm for EC2 CPU Utilization
resource "aws_cloudwatch_metric_alarm" "cpu_alarm" {
  alarm_name          = "High_CPU_Utilization"
  comparison_operator = "GreaterThanThreshold"
  evaluation_periods  = "1"
  metric_name        = "CPUUtilization"
  namespace          = "AWS/EC2"
  period             = "60"
  statistic          = "Average"
  threshold          = "80"  # Trigger if CPU > 80%
  alarm_description  = "This alarm fires if CPU utilization exceeds 80%."
 
  dimensions = {
    InstanceId = "i-0abcd1234efgh5678"  # Replace with your EC2 Instance ID
  }
 
  alarm_actions = ["arn:aws:sns:us-east-1:123456789012:your-sns-topic"]  # Replace with your SNS Topic ARN
  ok_actions    = ["arn:aws:sns:us-east-1:123456789012:your-sns-topic"]  # Same or different actions for OK state
  insufficient_data_actions = []
}
 
# Optional: Create an SNS topic for notifications
resource "aws_sns_topic" "example" {
  name = "example-sns-topic"
}
 
# Subscribe an email to the SNS topic
resource "aws_sns_topic_subscription" "example" {
  topic_arn = aws_sns_topic.example.arn
  protocol  = "email"
  endpoint  = "youremail@example.com"  # Replace with your email
}
